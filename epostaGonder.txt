Sub epostaGonder()
Dim OutlookApp As Object
    Dim MailItem As Object
    Dim ws As Worksheet
    Dim i As Integer
    
    ' Excel tablosunu seçme:
    Set ws = ThisWorkbook.Sheets("personel") ' Excel tablonuzun adını buraya yazın

    ' Outlook nesnesini tanımlama:
    Set OutlookApp = CreateObject("Outlook.Application")
    
    ' Exceldeki tüm satırları tek tek okuma ve eposta gönderme (Verilerin ikinci satırdan başladığı varsamıyla)
    i = 2
    While ws.Cells(i, 1).Value <> ""
        Set MailItem = OutlookApp.CreateItem(0)
        
        With MailItem
            .To = ws.Cells(i, 1).Value  	' A kolonundaki Göndereceğiniz eposta adresleri
            .CC = ws.Cells(i, 2).Value  	' B kolonundaki Bilgi (CC) olarak göndereceğiniz eposta adresleri
            .Subject = "Emeklilik Mesajı" 	' Konu olarak yazacağınız ifade
            .Body = "Sayın" 			' Hitap ifadesi _
            & vbNewLine & ws.Cells(i, 3).Value & "" 	' Göndereceğiniz kişinin adı ve soyadının olduğu C kolonu (3'üncü kolon) _
            & vbNewLine & ws.Cells(i, 4).Value & ""	' Göndereceğiniz kişinin ünvanı/mesleği/görevi vb. D kolonu (4'üncü kolon) _
            & vbNewLine & " " 				' Boşluk satırı _
            & vbNewLine & "63 yaşında bir büyüğünüz (abiniz, amcanız) olarak ..."  ' Birinci Paragraf _
            & vbNewLine & " " 	                         ' Boşluk satırı _
            & vbNewLine & "Hiç kimseye kırgın ve küskün değilim. ..." 	' İkinci Paragraf _
            & vbNewLine & " " _				' Boşluk satırı _
            & vbNewLine & "Sizlere de, benim gibi aileniz ve tüm ..." 	'Üçüncü Paragraf _
            & vbNewLine & " " _				' Boşluk satırı _
            & vbNewLine & "Dr. Hayati TAŞTAN" 		' Adınız ve Soyadınız _
            & vbNewLine & "Harita Yüksek Mühendisi" 	' Mesleğiniz _
            & vbNewLine & "Emekli Kıdemli Albay" 	' Göreviniz _
            & vbNewLine & "haytastan@gmail.com	 	' Göreviniz _
            & vbNewLine & " " 				' Boşluk satırı _
            & vbNewLine & "NOT:" 			' NOT _
            & vbNewLine & "Linkedin kullananlar aşağıdaki derleme makalemi okuyabilirler."    	' NOT Satırı -1 _
            & vbNewLine & "https://www.linkedin.com/posts/activity-7283067065681276928-aOdA/"   ' NOT Satırı -2 _
            & vbNewLine & "Bundan sonra beni Linkedin'den takip edebilirisiniz :)"		' NOT Satırı -3 _
            
            
            '.Display 					' Eğer göndermeden önce epostaları görmek isterseniz, satır başındaki ' sembolünü silin
	    .Send 					' Gönderme işlemi
        End With
        
        i = i + 1
    Wend
    
    ' Cleanup
    Set MailItem = Nothing
    Set OutlookApp = Nothing
End Sub
